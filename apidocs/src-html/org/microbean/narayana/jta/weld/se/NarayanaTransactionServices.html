017package org.microbean.narayana.jta.weld.se;

019import javax.enterprise.inject.Instance;

021import javax.enterprise.inject.spi.CDI;

023import javax.transaction.RollbackException;
024import javax.transaction.Status;
025import javax.transaction.Synchronization;
026import javax.transaction.SystemException;
027import javax.transaction.Transaction;
028import javax.transaction.TransactionManager;
029import javax.transaction.UserTransaction;

031import org.jboss.weld.transaction.spi.TransactionServices;

033/**
034 * A {@link TransactionServices} implementation that uses the <a
035 * href="https://narayana.io/">Narayana transaction engine</a> and
036 * does not use JNDI.
037 *
038 * <p>{@link TransactionServices} implementations are used by <a
039 * href="https://docs.jboss.org/weld/reference/latest/en-US/html/index.html">Weld</a>
040 * for <a
041 * href="http://docs.jboss.org/cdi/spec/2.0/cdi-spec.html#transactional_observer_methods">transactional
042 * observer notification</a> as well as for providing the
043 * implementation backing the <a
044 * href="http://docs.jboss.org/cdi/spec/2.0/cdi-spec.html#additional_builtin_beans">built-in
045 * {@code UserTransaction} CDI bean</a>.</p>
046 *
047 * @author <a href="https://about.me/lairdnelson"
048 * target="_parent">Laird Nelson</a>
049 *
050 * @see TransactionServices
051 */
052public class NarayanaTransactionServices implements TransactionServices {
053
054
055  /*
056   * Constructors.
057   */
058
059
060  /**
061   * Creates a new {@link NarayanaTransactionServices}.
062   */
063  public NarayanaTransactionServices() {
064    super();
065  }
066
067
068  /*
069   * Instance methods.
070   */

072
073  /**
074   * Returns the {@link UserTransaction} present in this environment
075   * by invoking the {@link
076   * com.arjuna.ats.jta.UserTransaction#userTransaction()} method and
077   * returning its result.
078   *
079   * <p>This method never returns {@code null}.</p>
080   *
081   * <p>The return value of this method is used as the backing
082   * implementation of the <a
083   * href="http://docs.jboss.org/cdi/spec/2.0/cdi-spec.html#additional_builtin_beans">built-in
084   * {@code UserTransaction} CDI bean</a>.</p>
085   *
086   * @return the non-{@code null} {@link UserTransaction} present in
087   * this environment
088   *
089   * @see com.arjuna.ats.jta.UserTransaction#userTransaction()
090   */
091  @Override
092  public final UserTransaction getUserTransaction() {
093    // We don't want to use, e.g.,
094    // CDI.current().select(UserTransaction.class).get() here because
095    // CDI containers like Weld are obliged per the specification to
096    // automatically provide a bean for UserTransaction.  Weld uses
097    // the return value of this method to create such a bean and we
098    // obviously need to avoid the infinite loop.
099    return com.arjuna.ats.jta.UserTransaction.userTransaction();
100  }
101
102  /**
103   * Returns {@code true} if the current {@link Transaction}
104   * {@linkplain Transaction#getStatus() has a status} indicating that
105   * it is active.
106   *
107   * <p>This method returns {@code true} if the current {@link
108   * Transaction} {@linkplain Transaction#getStatus() has a status}
109   * equal to one of the following values:</p>
110   *
111   * <ul>
112   *
113   * <li>{@link Status#STATUS_ACTIVE}</li>
114   *
115   * <li>{@link Status#STATUS_COMMITTING}</li>
116   *
117   * <li>{@link Status#STATUS_MARKED_ROLLBACK}</li>
118   *
119   * <li>{@link Status#STATUS_PREPARED}</li>
120   *
121   * <li>{@link Status#STATUS_PREPARING}</li>
122   *
123   * <li>{@link Status#STATUS_ROLLING_BACK}</li>
124   *
125   * </ul>
126   *
127   * @return {@code true} if the current {@link Transaction}
128   * {@linkplain Transaction#getStatus() has a status} indicating that
129   * it is active; {@code false} otherwise
130   *
131   * @exception RuntimeException if an invocation of the {@link
132   * Transaction#getStatus()} method resulted in a {@link
133   * SystemException}
134   *
135   * @see Status
136   */
137  @Override
138  public final boolean isTransactionActive() {
139    final boolean returnValue;
140    final Instance<Transaction> transactions = CDI.current().select(Transaction.class);
141    assert transactions != null;
142    if (!transactions.isUnsatisfied()) {
143      final Transaction transaction = transactions.get();
144      assert transaction != null;
145      boolean temp = false;
146      try {
147        final int status = transaction.getStatus();
148        temp =
149          status == Status.STATUS_ACTIVE ||
150          status == Status.STATUS_COMMITTING ||
151          status == Status.STATUS_MARKED_ROLLBACK ||
152          status == Status.STATUS_PREPARED ||
153          status == Status.STATUS_PREPARING ||
154          status == Status.STATUS_ROLLING_BACK;
155      } catch (final SystemException e) {
156        throw new RuntimeException(e.getMessage(), e);
157      } finally {
158        returnValue = temp;
159      }
160    } else {
161      returnValue = false;
162    }
163    return returnValue;
164  }
165
166  /**
167   * Registers the supplied {@link Synchronization} with the current
168   * {@link Transaction}.
169   *
170   * @exception RuntimeException if an invocation of the {@link
171   * TransactionManager#getTransaction()} method resulted in a {@link
172   * SystemException}, or if an invocation of the {@link
173   * Transaction#registerSynchronization(Synchronization)} method
174   * resulted in either a {@link SystemException} or a {@link
175   * RollbackException}
176   *
177   * @see Transaction#registerSynchronization(Synchronization)
178   */
179  @Override
180  public final void registerSynchronization(final Synchronization synchronization) {
181    final CDI<Object> cdi = CDI.current();    
182    final Instance<Transaction> transactionInstance = cdi.select(Transaction.class);
183    Transaction transaction = null;
184    if (transactionInstance.isUnsatisfied()) {
185      Instance<TransactionManager> transactionManagerInstance = cdi.select(TransactionManager.class);
186      assert transactionManagerInstance != null;
187      final TransactionManager transactionManager;
188      if (transactionManagerInstance.isUnsatisfied()) {
189        transactionManager = com.arjuna.ats.jta.TransactionManager.transactionManager();
190      } else {
191        transactionManager = transactionManagerInstance.get();
192      }
193      if (transactionManager != null) {
194        try {
195          transaction = transactionManager.getTransaction();
196        } catch (final SystemException e) {
197          throw new RuntimeException(e.getMessage(), e);
198        }
199      }
200    } else {
201      transaction = transactionInstance.get();
202    }
203    if (transaction != null) {
204      try {
205        transaction.registerSynchronization(synchronization);
206      } catch (final SystemException | RollbackException e) {
207        throw new RuntimeException(e.getMessage(), e);
208      }
209    }
210  }

214   * this object.

216  @Override
217  public synchronized final void cleanup() {
218
219  }
